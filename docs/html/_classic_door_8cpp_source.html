25void AClassicDoor::Interaction()
26{
27 GEngine->AddOnScreenDebugMessage(-1, 2.f, FColor::Green, FString::Printf(TEXT("Interacted")));
28 if (IsClosed) {
29 SetOnSameSide();
30 Timeline.Play();
31 GEngine->AddOnScreenDebugMessage(-1, 2.f, FColor::Green, FString::Printf(TEXT("Opening")));
32 Sound1->Play();
33 }
34 else {
35 GEngine->AddOnScreenDebugMessage(-1, 2.f, FColor::Green, FString::Printf(TEXT("Closing")));
36 Timeline.Reverse();
37 Sound2->Play();
38 }
39 IsClosed = !IsClosed;
40}
