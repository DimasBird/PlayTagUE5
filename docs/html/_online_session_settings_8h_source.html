495 bool GetJoinability(bool& bPublicJoinable, bool& bFriendJoinable, bool& bInviteOnly, bool& bAllowInvites) const
496 {
497 // Only states that have a valid session are considered
498 if (SessionState != EOnlineSessionState::NoSession && SessionState != EOnlineSessionState::Creating && SessionState != EOnlineSessionState::Destroying)
499 {
500 bool bAllowJIP = SessionSettings.bAllowJoinInProgress || (SessionState != EOnlineSessionState::Starting && SessionState != EOnlineSessionState::InProgress);
501 if (bAllowJIP)
502 {
503 bPublicJoinable = SessionSettings.bShouldAdvertise || SessionSettings.bAllowJoinViaPresence;
504 bFriendJoinable = SessionSettings.bAllowJoinViaPresenceFriendsOnly;
505 bInviteOnly = !bPublicJoinable && !bFriendJoinable && SessionSettings.bAllowInvites;
506 bAllowInvites = SessionSettings.bAllowInvites;
507 }
508 else
509 {
510 bPublicJoinable = false;
511 bFriendJoinable = false;
512 bInviteOnly = false;
513 bAllowInvites = false;
514 }
515
516 // Valid session, joinable or otherwise
517 return true;
518 }
519
520 // Invalid session
521 return false;
522 }
